109 inline int GEMAmplification::getStackAmplification(const CRU& cru, const PadPos& pos, const AmplificationMode mode, int nElectrons)
110 {
113  switch (mode) {
114  case AmplificationMode::FullMode: {
115  return static_cast<int>(static_cast<float>(getStackAmplification(nElectrons)) *
116  mGainMap->getValue(cru, pos.getRow(), pos.getPad()));
117  break;
118  }
119  case AmplificationMode::EffectiveMode: {
120  return static_cast<int>(static_cast<float>(getEffectiveStackAmplification(nElectrons)) *
121  mGainMap->getValue(cru, pos.getRow(), pos.getPad()));
122  break;
123  }
124  }
125  return nElectrons;
