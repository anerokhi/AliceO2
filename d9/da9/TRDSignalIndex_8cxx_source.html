167 {

170  //

188  mSortedIndex = new RowCol[mMaxLimit + 1];

191 
192  resetArrays();
193  resetCounters();
194 
195  mCountRC = 1;
196 }

200 {

299 bool TRDSignalIndex::checkSorting(Int_t& row, Int_t& col)
300 {
301  //
302  // Check whether array was read to end or it was not sorted until now
303  //
304 
305  if (mSortedWasInit || mCountRC == 1) { //we already reached the end of the array
306  resetCounters();
307  row = mCurrRow;
308  col = mCurrCol;
309  return kFALSE;
310  } else { //we have not sorted the array up to now, let's do so
311  initSortedIndex();
312  return nextRCIndex(row, col);
313  }
314 }

317 bool TRDSignalIndex::nextRCTbinIndex(Int_t& row, Int_t& col, Int_t& tbin)
318 {
319  //
320  // Returns the next tbin, or if there is no next time bin, it returns the
321  // next used RC combination.
322  //
323 
324  if (nextTbinIndex(tbin)) {
325  row = mCurrRow;
326  col = mCurrCol;
327  return kTRUE;
328  } else {
329  if (nextRCIndex(row, col)) {
330  return nextRCTbinIndex(row, col, tbin);
331  }
332  }
333 
334  return kFALSE;
335 }
