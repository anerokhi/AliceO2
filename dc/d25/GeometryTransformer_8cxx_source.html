45 ROOT::Math::Transform3D getDefaultRPCTransform(bool isRight, int chamber, int rpc)
46 {
48  const double degToRad = TMath::DegToRad();
49  double angle = isRight ? 0. : 180.;
50  ROOT::Math::Rotation3D rot(ROOT::Math::RotationY(angle * degToRad));
51  double xSign = isRight ? 1. : -1.;
52  double xPos = xSign * Constants::getRPCCenterPosX(chamber, rpc);
53  double sign = (rpc % 2 == 0) ? 1. : -1;
54  if (!isRight) {
55  sign *= -1.;
56  }
57  double zPos = sign * Constants::sRPCZShift;
58  double newZ = Constants::sDefaultChamberZ[0] + zPos;
59  double oldZ = Constants::sDefaultChamberZ[0] - zPos;
60  double yPos = Constants::getRPCHalfHeight(chamber) * (rpc - 4) * (1. + newZ / oldZ);
61  ROOT::Math::Translation3D trans(xPos, yPos, zPos);
62  return trans * rot;
63 }

65 GeometryTransformer createDefaultTransformer()
66 {
68  GeometryTransformer geoTrans;
69  for (int ich = 0; ich < Constants::sNChambers; ++ich) {
70  for (int iside = 0; iside < 2; ++iside) {
71  bool isRight = (iside == 0);
72  for (int irpc = 0; irpc < Constants::sNRPCLines; ++irpc) {
73  int deId = Constants::getDEId(isRight, ich, irpc);
74  ROOT::Math::Transform3D matrix = getDefaultChamberTransform(ich) * getDefaultRPCTransform(isRight, ich, irpc);
75  geoTrans.setMatrix(deId, matrix);
76  }
77  }
78  }
79  return geoTrans;
80 }
