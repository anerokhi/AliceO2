94  void* do_allocate(std::size_t bytes, std::size_t alignment) override
95  {
96  if (initialImport) {
97  if (bytes > mMessageSize) {
98  throw std::bad_alloc();
99  }
100  initialImport = false;
101  return mMessageData;
102  } else {
103  return mUpstream->allocate(bytes, alignment);
104  }
105  }
106  void do_deallocate(void* p, std::size_t bytes, std::size_t alignment) override
107  {
108  mUpstream->deallocate(p, bytes, alignment);
109  return;
110  }
111  bool do_is_equal(const memory_resource& other) const noexcept override
112  {
113  // since this uniquely owns the message it can never be equal to anybody else
114  return false;
115  }

119 // This in general (as in STL) is a bad idea, but here it is safe to inherit from an allocator since we
120 // have no additional data and only override some methods so we don't get into slicing and other problems.
121 template <typename T>
122 class SpectatorAllocator : public boost::container::pmr::polymorphic_allocator<T>
123 {
124  public:
125  using boost::container::pmr::polymorphic_allocator<T>::polymorphic_allocator;
126 
127  // skip default construction of empty elements
128  // this is important for two reasons: one: it allows us to adopt an existing buffer (e.g. incoming message) and
129  // quickly construct large vectors while skipping the element initialization.
130  template <class U>
131  void construct(U*)
132  {
133  }
134 
135  // dont try to call destructors, makes no sense since resource is managed externally AND allowed
136  // types cannot have side effects
137  template <typename U>
138  void destroy(U*)
139  {
140  }
141 
142  T* allocate(size_t size) { return reinterpret_cast<T*>(this->resource()->allocate(size * sizeof(T), 0)); }
143  void deallocate(T* ptr, size_t size)
144  {
145  this->resource()->deallocate(const_cast<typename std::remove_cv<T>::type*>(ptr), size);
146  }
147 };
