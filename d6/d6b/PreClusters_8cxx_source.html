64 {
66  std::vector<int> neighbours;
67  if (icolumn == 6) {
68  return neighbours;
69  }
70  PreClusterBP& pcB = mPreClustersBP[icolumn][idx];

72  PreClusterBP& neigh = mPreClustersBP[icolumn + 1][idx];
73  if (neigh.area.getYmin() > pcB.area.getYmax()) {
