124 inline bool DigitGlobalPad::compareMClabels(const MCCompLabel& label1, const MCCompLabel& label2) const
125 {
126  // we compare directly on the bare label (in which eventID, labelID etc. are encoded)
127  // this avoids any logical operator on the label; optimization motivated from an Intel VTune analysis
128  // (note that this is also faster than using the operator= of MCCompLabel)
129  return label1.getRawValue() == label2.getRawValue();
130 }

132 template <DigitzationMode MODE>
133 inline void DigitGlobalPad::fillOutputContainer(std::vector<Digit>& output,
134  dataformats::MCTruthContainer<MCCompLabel>& mcTruth,
135  const CRU& cru, TimeBin timeBin,
136  GlobalPadNumber globalPad,
137  o2::dataformats::LabelContainer<std::pair<MCCompLabel, int>, false>& labels,
138  float commonMode)
139 {
140  const static Mapper& mapper = Mapper::instance();
141  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
142  const PadPos pad = mapper.padPos(globalPad);
143  static std::vector<std::pair<MCCompLabel, int>> labelCollector; // static workspace container for sorting
144 
147  const float totalADC = mChargePad - commonMode; // common mode is subtracted here in order to properly apply noise,
148  // pedestals and saturation of the SAMPA
149 
150  float noise, pedestal;
151  const float mADC = sampaProcessing.makeSignal<MODE>(totalADC, cru.sector(), globalPad, pedestal, noise);
152 
154  if (mADC > 0 && mChargePad > 0) {
155  auto labelview = labels.getLabels(mID);
156 
158  const auto digiPos = output.size();
159  output.emplace_back(cru, mADC, pad.getRow(), pad.getPad(), timeBin);
160 
161  labelCollector.clear();
162  for (auto& mcLabel : labelview) {
163  labelCollector.push_back(mcLabel);
164  }
165  if (labelview.size() > 1) {
167  using P = std::pair<MCCompLabel, int>;
168  std::sort(labelCollector.begin(), labelCollector.end(), [](const P& a, const P& b) { return a.second > b.second; });
169  }
170  for (auto& mcLabel : labelCollector) {
171  mcTruth.addElement(digiPos, mcLabel.first);
172  }
173  }
174 }
