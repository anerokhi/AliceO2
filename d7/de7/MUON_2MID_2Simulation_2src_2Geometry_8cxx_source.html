139  assertMedium(Medium::Bakelite));
140

143  new TGeoBBox(Form("%sInsuBox", name), halfLength, halfHeight, kInsulatorHalfThickness),
144  assertMedium(Medium::Mylar));
145 
147  // cooper foil
148  auto copper = new TGeoVolume(Form("Copper %s", name),
149  new TGeoBBox(Form("%sCopperBox", name), halfLength, halfHeight, kCopperHalfThickness),
150  assertMedium(Medium::Copper));
151 
152  // mylar foil
153  auto mylar = new TGeoVolume(Form("Mylar %s", name),
154  new TGeoBBox(Form("%sMylarBox", name), halfLength, halfHeight, kMylarHalfThickness),
155  assertMedium(Medium::Mylar));
156 
157  // styrofoam plane
158  auto styro = new TGeoVolume(Form("Styrofoam %s", name),
159  new TGeoBBox(Form("%sStyroBox", name), halfLength, halfHeight, kStyrofoamHalfThickness),
160  assertMedium(Medium::Styrofoam));
161 
163  // aluminium foil
164  auto alu = new TGeoVolume(Form("Aluminium %s", name),
165  new TGeoBBox(Form("%sAluBox", name), halfLength, halfHeight, kAluminiumHalfThickness),
166  assertMedium(Medium::Aluminium));
167 
168  // nomex
169  auto nomex = new TGeoVolume(Form("Nomex %s", name),
170  new TGeoBBox(Form("%sNomexBox", name), halfLength, halfHeight, kNomexHalfThickness),
171  assertMedium(Medium::Nomex));
172 
173  // change the volume shape if we are creating a "cut" RPC
174  if (type == RPCtype::TopCut || type == RPCtype::BottomCut) {
175  // dimensions of the cut
176  double cutHalfLength = Constants::getLocalBoardWidth(iChamber) / 2.;
177  double cutHalfHeight = Constants::getLocalBoardHeight(iChamber) / 2.;
178 
179  bool isTopCut = (type == RPCtype::TopCut);
180  const char* cutName = Form("%sCut%s", (isTopCut) ? "top" : "bottom", name);
181 
182  // position of the cut w.r.t the center of the RPC
183  auto cutPos = new TGeoTranslation(Form("%sPos", cutName), cutHalfLength - halfLength, (isTopCut) ? halfHeight - cutHalfHeight : cutHalfHeight - halfHeight, 0.);
184  cutPos->RegisterYourself();
185 
186  // for each volume, create a box and change the volume shape by extracting the cut shape
187  new TGeoBBox(Form("%sGasCut", name), cutHalfLength, cutHalfHeight, kGasHalfThickness);
188  gas->SetShape(new TGeoCompositeShape(Form("%sGasShape", name), Form("%sGasBox-%sGasCut:%sPos", name, name, cutName)));
189 
190  new TGeoBBox(Form("%sElecCut", name), cutHalfLength, cutHalfHeight, kElectrodHalfThickness);
191  electrod->SetShape(new TGeoCompositeShape(Form("%sElecShape", name), Form("%sElecBox-%sElecCut:%sPos", name, name, cutName)));
192 
193  new TGeoBBox(Form("%sInsuCut", name), cutHalfLength, cutHalfHeight, kInsulatorHalfThickness);
194  insu->SetShape(new TGeoCompositeShape(Form("%sInsuShape", name), Form("%sInsuBox-%sInsuCut:%sPos", name, name, cutName)));
195 
196  new TGeoBBox(Form("%sCopperCut", name), cutHalfLength, cutHalfHeight, kCopperHalfThickness);
197  copper->SetShape(new TGeoCompositeShape(Form("%sCopperShape", name), Form("%sCopperBox-%sCopperCut:%sPos", name, name, cutName)));
198 
199  new TGeoBBox(Form("%sMylarCut", name), cutHalfLength, cutHalfHeight, kMylarHalfThickness);
200  mylar->SetShape(new TGeoCompositeShape(Form("%sMylarShape", name), Form("%sMylarBox-%sMylarCut:%sPos", name, name, cutName)));
201 
202  new TGeoBBox(Form("%sStyroCut", name), cutHalfLength, cutHalfHeight, kStyrofoamHalfThickness);
203  styro->SetShape(new TGeoCompositeShape(Form("%sStyroShape", name), Form("%sStyroBox-%sStyroCut:%sPos", name, name, cutName)));
204 
205  new TGeoBBox(Form("%sAluCut", name), cutHalfLength, cutHalfHeight, kAluminiumHalfThickness);

207

233  z += halfThickness;
234 
235  halfThickness = kMylarHalfThickness;
236  z += halfThickness;
237  rpc->AddNode(mylar, 1, new TGeoTranslation(0., 0., z));
238  rpc->AddNode(mylar, 2, new TGeoTranslation(0., 0., -z));

244  rpc->AddNode(styro, 2, new TGeoTranslation(0., 0., -z));

279 TGeoMatrix* getTransformation(const ROOT::Math::Transform3D& matrix)
280 {
282  double xx, xy, xz, dx, yx, yy, yz, dy, zx, zy, zz, dz;
283  matrix.GetComponents(xx, xy, xz, dx, yx, yy, yz, dy, zx, zy, zz, dz);
284  double vect[3] = { dx, dy, dz };
285  double rotMatrix[9] = { xx, xy, xz, yx, yy, yz, zx, zy, zz };
286  TGeoHMatrix* geoMatrix = new TGeoHMatrix("Transformation");
287  geoMatrix->SetTranslation(vect);
288  geoMatrix->SetRotation(rotMatrix);
289  return geoMatrix;
290 }

314 TGeoVolume* createChamber(int iChamber)
315 {
317 
318  auto chamber = new TGeoVolumeAssembly(getChamberVolumeName(iChamber).c_str());
319 
320  double scale = Constants::sScaleFactors[iChamber];
321 
322  // create the service volumes
323  auto vertSupp = createVerticalSupport(iChamber);
324  auto horiSupp = createHorizontalSupport(iChamber);
325 
326  // create the 4 types of RPC
327  auto longRPC = createRPC(RPCtype::Long, iChamber);
328  auto bottomCutRPC = createRPC(RPCtype::BottomCut, iChamber);
329  auto topCutRPC = createRPC(RPCtype::TopCut, iChamber);
330  auto shortRPC = createRPC(RPCtype::Short, iChamber);
331 
332  // for node counting
333  int iHoriSuppNode = 0, iVertSuppNode = 0;
334 
335  // place the volumes on both side of the chamber
336  for (int iside = 0; iside < 2; iside++) {
337 
338  bool isRight = (iside == 0);
339  double xSign = (isRight) ? 1. : -1.;
340 
341  // place 4 vertical supports per side
342  for (int i = 0; i < 4; i++) {
343  chamber->AddNode(vertSupp, iVertSuppNode++, new TGeoTranslation(xSign * kVerticalSupportXPos[i] * scale, 0., 0.));
344  }
345 
346  // place the RPCs
347  for (int iRPC = 0; iRPC < Constants::sNRPCLines; iRPC++) {
348 
349  double x = xSign * Constants::getRPCCenterPosX(iChamber, iRPC);
350  double zSign = (iRPC % 2 == 0) ? 1. : -1.;
351 
352  if (!isRight) {
353  zSign *= -1.;
354  }
355  double z = zSign * Constants::sRPCZShift;
356  double y = 2 * Constants::getRPCHalfHeight(iChamber) * (iRPC - 4) / (1 - (z / Constants::sDefaultChamberZ[0]));
357 
358  // ID convention (from bottom to top of the chamber) : long, long, long, cut, short, cut, long, long, long
359  TGeoVolume* rpc = nullptr;
360  switch (iRPC) {
361  case 4: // short
362  rpc = shortRPC;
363  break;
364  case 5: // cut (bottom)
365  rpc = bottomCutRPC;
366  break;
367  case 3: // cut (top)
368  rpc = topCutRPC;
369  break;
370  default: // long
371  rpc = longRPC;
372  break;
373  }
374 
375  int deId = Constants::getDEId(isRight, iChamber, iRPC);
376  chamber->AddNode(rpc, deId, getTransformation(getDefaultRPCTransform(isRight, iChamber, iRPC)));
377 
378  // place 3 horizontal supports behind the RPC (and the vertical rods)
379  x = xSign * kHorizontalSupportPos[0] * scale;
380  z = -zSign * kHorizontalSupportPos[2];
381  for (int i = 0; i < 3; i++) {
382  chamber->AddNode(horiSupp, iHoriSuppNode++, new TGeoTranslation(x, y + (i - 1) * kHorizontalSupportPos[1] * scale, z));
383  }
384 
385  } // end of the loop over the number of RPC lines
386 
387  } // end of the side loop
388 
389  return chamber;
390 }
