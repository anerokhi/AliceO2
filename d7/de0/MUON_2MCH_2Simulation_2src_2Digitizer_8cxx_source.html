99 int Digitizer::processHit(const Hit& hit, int detID, double event_time, int labelIndex)
100 {
101  Point3D<float> pos(hit.GetX(), hit.GetY(), hit.GetZ());
102 
103  Response& resp = response(isStation1(detID));
104 
105  //convert energy to charge
106  auto charge = resp.etocharge(hit.GetEnergyLoss());
107  auto time = hit.GetTime(); //to be used for pile-up
108 
109  //transformation from global to local
110  auto t = o2::mch::getTransformation(detID, *gGeoManager);
111  Point3D<float> lpos;
112  t.MasterToLocal(pos, lpos);
113 
114  auto anodpos = resp.getAnod(lpos.X());
115  auto fracplane = resp.chargeCorr();
116  auto chargebend = fracplane * charge;
117  auto chargenon = charge / fracplane;
118 
119  //borders of charge gen.
120  auto xMin = anodpos - resp.getQspreadX() * 0.5;
121  auto xMax = anodpos + resp.getQspreadX() * 0.5;
122  auto yMin = lpos.Y() - resp.getQspreadY() * 0.5;
123  auto yMax = lpos.Y() + resp.getQspreadY() * 0.5;
124 
125  //get segmentation for detector element
126  auto& seg = segmentation(detID);
127  auto localX = anodpos;
128  auto localY = lpos.Y();
129 
130  //get area for signal induction from segmentation
131  //single pad as check
132  int padidbendcent = 0;
133  int padidnoncent = 0;
134  bool padexists = seg.findPadPairByPosition(localX, localY, padidbendcent, padidnoncent);
135  if (!padexists) {
136  LOG(ERROR) << "Did not find _any_ pad for localX,Y=" << localX << "," << localY;
137  return 0;
138  }
139 
140  seg.forEachPadInArea(xMin, yMin, xMax, yMax, [&resp, &digits = this->mDigits, chargebend, chargenon, localX, localY, &seg, labelIndex ](int padid) {
141  auto dx = seg.padSizeX(padid) * 0.5;
142  auto dy = seg.padSizeY(padid) * 0.5;
143  auto xmin = (localX - seg.padPositionX(padid)) - dx;
144  auto xmax = xmin + dx;
145  auto ymin = (localY - seg.padPositionY(padid)) - dy;
146  auto ymax = ymin + dy;
147  auto q = resp.chargePadfraction(xmin, xmax, ymin, ymax);
148  if (seg.isBendingPad(padid)) {
149  q *= chargebend;
150  } else {
151  q *= chargenon;
152  }
153  auto signal = resp.response(q);
154  digits.emplace_back(padid, signal, labelIndex);
155  });
156  return mDigits.size();
157 }

160 void Digitizer::fillOutputContainer(std::vector<Digit>& digits)
161 {
162  // filling the digit container
163  if (mDigits.empty())
164  return;
165 
166  auto itBeg = mDigits.begin();
167  auto iter = itBeg;
168  for (; iter != mDigits.end(); ++iter) {
169  digits.emplace_back(*iter);
170  }
171  mDigits.erase(itBeg, iter);
172  mMCTruthOutputContainer.clear();
173  for (int index = 0; index < mMCTruthContainer.getIndexedSize(); ++index) {
174  mMCTruthOutputContainer.addElements(index, mMCTruthContainer.getLabels(index));
175  }
176 }
