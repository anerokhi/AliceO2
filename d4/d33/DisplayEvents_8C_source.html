165  qclus->RefitPlex();
166  TEveTrans& ct = qclus->RefMainTrans();
167  ct.RotateLF(1, 3, 0.5 * TMath::Pi());
168  ct.SetPos(0, 0, 0);

185 TEveElement* Data::getEveTracks(const std::vector<Cluster>& clus, const std::vector<o2::ITS::TrackITS>& trks)
186 {

214  }
