100 void TRDArrayDictionary::allocate(int nrow, int ncol, int ntime)
101 {
102  //
103  // Allocates memory for the dictionary array with dimensions
104  // Row*NumberOfNecessaryMCMs*ADCchannelsInMCM*Time
105  // To be consistent with TRDArrayADC
106  // Object initialized to -1
107  //
108 
109  mNrow = nrow;
110  mNcol = ncol;
111  mNtime = ntime;
112  int adcchannelspermcm = TRDFeeParam::getNadcMcm();
113  int padspermcm = TRDFeeParam::getNcolMcm();
114  int numberofmcms = mNcol / padspermcm;
115  mNumberOfChannels = numberofmcms * adcchannelspermcm;
116  mNDdim = nrow * mNumberOfChannels * ntime;
117  if (mDictionary.size() != mNDdim) {
118  mDictionary.resize(mNDdim);
119  }
120  memset(&mDictionary[0], -1, sizeof(mDictionary[0]) * mNDdim);
121 }

124 void TRDArrayDictionary::compress()
125 {
126  //
127  // Compress the array
128  //
129 
130  int counter = 0;
131  int newDim = 0;
132  int j;
133  int r = 0;
134  int k = 0;
135 
136  std::vector<int> longArr(mNDdim); //do not change to bool
137 
138  memset(&longArr[0], 0, sizeof(longArr[0]) * mNDdim);
139 
140  for (int i = 0; i < mNDdim; i++) {
141  j = 0;
142  if (mDictionary[i] == -1) {
143  for (k = i; k < mNDdim; k++) {
144  if (mDictionary[k] == -1) {
145  j = j + 1;
146  longArr[r] = j;
147  } else {
148  break;
149  }
150  }
151  r = r + 1;
152  }
153  i = i + j;
154  }
155 
156  //Calculate the size of the compressed array
157  for (int i = 0; i < mNDdim; i++) {
158  if (longArr[i] != 0) {
159  counter = counter + longArr[i] - 1;
160  }
161  }
162  newDim = mNDdim - counter; //Size of the compressed array
163 
164  //Fill the buffer of the compressed array
165  std::vector<int> buffer(newDim);
166  int counterTwo = 0;
167  int g = 0;
168  for (int i = 0; i < newDim; i++) {
169  if (counterTwo < mNDdim) {
170  if (mDictionary[counterTwo] != -1) {
171  buffer[i] = mDictionary[counterTwo];
172  }
173  if (mDictionary[counterTwo] == -1) {
174  buffer[i] = -(longArr[g]);
175  counterTwo = counterTwo + longArr[g] - 1;
176  g++;
177  }
178  counterTwo++;
179  }
180  }
181 
182  mDictionary = buffer;
183  mNDdim = newDim;
184 
185  mFlag = kFALSE; // This way it can be expanded afterwards
186 }

189 void TRDArrayDictionary::expand()
190 {
191  //
192  // Expand the array
193  //
194 
195  if (mNDdim == 0) {
196  LOG(error) << "Called expand with dinesion of zero ";
197  return;
198  }
199 
200  int dimexp = 0;
201 
202  // if(WasExpandCalled())
203  // return;
204 
205  if (mNDdim == mNrow * mNumberOfChannels * mNtime)
206  return;
207 
208  if (mNDdim == 1) {
209  dimexp = -mDictionary[0];
210  mDictionary.resize(dimexp);
211  mNDdim = dimexp;
212  // Re-initialize the array
213  memset(&mDictionary[0], -1, sizeof(int) * dimexp);
214  mFlag = kTRUE; // Not expand again
215  return;
216  }
217 
218  std::vector<int> longArr(mNDdim);
219 
220  //Initialize the array
221  memset(&longArr[0], 0, sizeof(longArr[0]) * mNDdim);
222 
223  int r2 = 0;
224  for (int i = 0; i < mNDdim; i++) {
225  if ((mDictionary[i] < 0) && (mDictionary[i] != -1)) {
226  longArr[r2] = -mDictionary[i];
227  r2++;
228  }
229  }
230 
231  //Calculate new dimensions
232  for (int i = 0; i < mNDdim; i++) {
233  if (longArr[i] != 0) {
234  dimexp = dimexp + longArr[i] - 1;
235  }
236  if (longArr[i] == 0) {
237 
238  break;
239  }
240  }
241  dimexp = dimexp + mNDdim;
242 
243  //Write in the buffer the new array
244  int contaexp = 0;
245  int h = 0;
246  std::vector<int> buffer(dimexp);
247 
248  memset(&buffer[0], -1, sizeof(buffer[0]) * dimexp);
249 
250  for (int i = 0; i < dimexp; i++) {
251  if (mDictionary[contaexp] >= -1) {
252  buffer[i] = mDictionary[contaexp];
253  }
254  if (mDictionary[contaexp] < -1) {
255  i = i + longArr[h] - 1;
256  h++;
257  }
258  contaexp++;
259  }
260 
261  //Copy the buffer
262  mDictionary = buffer;
263  mNDdim = dimexp;
264  mFlag = kTRUE; // Not expand again
265 }
