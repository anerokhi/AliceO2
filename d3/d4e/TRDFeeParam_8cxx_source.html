157 int TRDFeeParam::getPadColFromADC(int irob, int imcm, int iadc) const
158 {
159  //
160  // Return which pad is connected to this adc channel.
161  //
162  // Return virtual pad number even if ADC is outside chamber
163  // to keep compatibility of data processing at the edge MCM.
164  // User has to check that this is in the chamber if it is essential.
165  // Return -100 if iadc is invalid.
166  //
167  // Caution: ADC ordering in the online data is opposite to the pad column ordering.
168  // And it is not one-by-one correspondence. Precise drawing can be found in:
169  // http://wiki.kip.uni-heidelberg.de/ti/TRD/index.php/Image:ROB_MCM_numbering.pdf
170  //
171 
172  if (iadc < 0 || iadc > mgkNadcMcm)
173  return -100;
174  int mcmcol = imcm % mgkNmcmRobInCol + getRobSide(irob) * mgkNmcmRobInCol; // MCM column number on ROC [0..7]
175  int padcol = mcmcol * mgkNcolMcm + mgkNcolMcm + 1 - iadc;
176  if (padcol < 0 || padcol >= mgkNcol)
177  return -1; // this is commented because of reason above OK
178 
179  return padcol;
180 }

183 int TRDFeeParam::getExtendedPadColFromADC(int irob, int imcm, int iadc) const
184 {
185  //
186  // Return which pad coresponds to the extended digit container pad numbering
187  // Extended digit container is designed to store all pad data including shared pad,
188  // so we have to introduce new virtual pad numbering scheme for this purpose.
189  //
190 
191  if (iadc < 0 || iadc > mgkNadcMcm)
192  return -100;
193  int mcmcol = imcm % mgkNmcmRobInCol + getRobSide(irob) * mgkNmcmRobInCol; // MCM column number on ROC [0..7]

195

214 int TRDFeeParam::getMCMfromSharedPad(int irow, int icol) const
215 {
216  //
217  // Return on which MCM this pad is directry connected.
218  // Return -1 for error.
219  //
220 
221  if (irow < 0 || icol < 0 || irow > mgkNrowC1 || icol > mgkNcol + 8 * 3)
222  return -1;
223 
224  int adc = 20 - (icol % 18) - 1;
225  switch (adc) {
226  case 2:
227  icol += 5;
228  break;
229  case 18:
230  icol -= 5;
231  break;
232  case 19:
233  icol -= 5;
234  break;
235  default:
236  icol += 0;
237  break;
238  }
239 
240  return (icol % (mgkNcol / 2)) / mgkNcolMcm + mgkNmcmRobInCol * (irow % mgkNmcmRobInRow);
241 }

300 int TRDFeeParam::extAliToAli(unsigned int dest, unsigned short linkpair, unsigned short rocType, int* mcmList, int listSize)
301 {
302  // Converts an extended ALICE ID which identifies a single MCM or a group of MCMs to
303  // the corresponding list of MCMs. Only broadcasts (127) are encoded as 127
304  // The return value is the number of MCMs in the list
305 
306  mcmList[0] = -1;
307 
308  short nmcm = 0;
309  unsigned int mcm, rob, robAB;
310  unsigned int cmA = 0, cmB = 0; // Chipmask for each A and B side
311 
312  // Default chipmask for 4 linkpairs (each bit correponds each alice-mcm)
313  static const unsigned int gkChipmaskDefLp[4] = { 0x1FFFF, 0x1FFFF, 0x3FFFF, 0x1FFFF };
314 
315  rob = dest >> 7; // Extract ROB pattern from dest.
316  mcm = dest & 0x07F; // Extract MCM pattern from dest.
317  robAB = getRobAB(rob, linkpair); // Get which ROB sides are selected.
318 
319  // Abort if no ROB is selected
320  if (robAB == 0) {
321  return 0;
322  }
323 
324  // Special case
325  if (mcm == 127) {
326  if (robAB == 3) { // This is very special 127 can stay only if two ROBs are selected
327  mcmList[0] = 127; // broadcase to ALL
328  mcmList[1] = -1;
329  return 1;
330  }
331  cmA = cmB = 0x3FFFF;
332  } else if ((mcm & 0x40) != 0) { // If top bit is 1 but not 127, this is chip group.
333  if ((mcm & 0x01) != 0) {
334  cmA |= 0x04444;
335  cmB |= 0x04444;
336  } // chip_cmrg
337  if ((mcm & 0x02) != 0) {
338  cmA |= 0x10000;
339  cmB |= 0x10000;
340  } // chip_bmrg
341  if ((mcm & 0x04) != 0 && rocType == 0) {
342  cmA |= 0x20000;
343  cmB |= 0x20000;
344  } // chip_hm3
345  if ((mcm & 0x08) != 0 && rocType == 1) {
346  cmA |= 0x20000;
347  cmB |= 0x20000;
348  } // chip_hm4
349  if ((mcm & 0x10) != 0) {
350  cmA |= 0x01111;
351  cmB |= 0x08888;
352  } // chip_edge
353  if ((mcm & 0x20) != 0) {
354  cmA |= 0x0aaaa;
355  cmB |= 0x03333;
356  } // chip_norm
357  } else { // Otherwise, this is normal chip ID, turn on only one chip.
358  cmA = 1 << mcm;
359  cmB = 1 << mcm;
360  }
361 
362  // Mask non-existing MCMs
363  cmA &= gkChipmaskDefLp[linkpair];
364  cmB &= gkChipmaskDefLp[linkpair];
365  // Remove if only one side is selected
366  if (robAB == 1)
367  cmB = 0;
368  if (robAB == 2)
369  cmA = 0;
370  if (robAB == 4 && linkpair != 2)
371  cmA = cmB = 0; // Restrict to only T3A and T3B
372 
373  // Finally convert chipmask to list of slaves
374  nmcm = chipmaskToMCMlist(cmA, cmB, linkpair, mcmList, listSize);
375 
376  return nmcm;
377 }

379 short TRDFeeParam::getRobAB(unsigned short robsel, unsigned short linkpair)
380 {
381  // Converts the ROB part of the extended ALICE ID to robs
382 
383  if ((robsel & 0x8) != 0) { // 1000 .. direct ROB selection. Only one of the 8 ROBs are used.
384  robsel = robsel & 7;
385  if ((robsel % 2) == 0 && (robsel / 2) == linkpair)
386  return 1; // Even means A side (position 0,2,4,6)
387  if ((robsel % 2) == 1 && (robsel / 2) == linkpair)
388  return 2; // Odd means B side (position 1,3,5,7)
389  return 0;
390  }
391 
392  // ROB group
393  if (robsel == 0) {
394  return 3;
395  } // Both ROB
396  if (robsel == 1) {
397  return 1;
398  } // A-side ROB
399  if (robsel == 2) {
400  return 2;
401  } // B-side ROB
402  if (robsel == 3) {
403  return 3;
404  } // Both ROB
405  if (robsel == 4) {
406  return 4;
407  } // Only T3A and T3B
408  // Other number 5 to 7 are ignored (not defined)
409 
410  return 0;
411 }

413 short TRDFeeParam::chipmaskToMCMlist(unsigned int cmA, unsigned int cmB, unsigned short linkpair, int* mcmList, int listSize)
414 {
415  // Converts the chipmask to a list of MCMs
416 
417  short nmcm = 0;
418  short i;
419  for (i = 0; i < 18; i++) { // 18: number of MCMs on a ROB
420  if ((cmA & (1 << i)) != 0 && nmcm < listSize) {
421  mcmList[nmcm] = ((linkpair * 2) << 7) | i;
422  ++nmcm;
423  }
424  if ((cmB & (1 << i)) != 0 && nmcm < listSize) {
425  mcmList[nmcm] = ((linkpair * 2 + 1) << 7) | i;
426  ++nmcm;
427  }
428  }
429 
430  mcmList[nmcm] = -1;
431  return nmcm;
432 }
