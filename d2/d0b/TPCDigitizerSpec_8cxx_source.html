82  void init(framework::InitContext& ic)
83  {
84  LOG(INFO) << "Initializing TPC digitization";
85 
87  auto& cdb = o2::TPC::CDBInterface::instance();
88  cdb.setUseDefaults();
89 
90  auto useDistortions = ic.options().get<int>("distortionType");
91  auto gridSizeString = ic.options().get<std::string>("gridSize");
92  auto triggeredMode = ic.options().get<bool>("TPCtriggered");
93 
94  std::vector<int> gridSize;
95  std::stringstream ss(gridSizeString);
96  while (ss.good()) {
97  std::string substr;
98  getline(ss, substr, ',');
99  gridSize.push_back(std::stoi(substr));
100  }
101  auto inputHistoString = ic.options().get<std::string>("initialSpaceChargeDensity");
102  std::vector<std::string> inputHisto;
103  std::stringstream ssHisto(inputHistoString);
104  while (ssHisto.good()) {
105  std::string substr;
106  getline(ssHisto, substr, ',');
107  inputHisto.push_back(substr);
108  }
109  if (useDistortions > 0) {
110  o2::TPC::SpaceCharge::SCDistortionType distortionType = useDistortions == 1 ? o2::TPC::SpaceCharge::SCDistortionType::SCDistortionsRealistic : o2::TPC::SpaceCharge::SCDistortionType::SCDistortionsConstant;
111  std::unique_ptr<TH3> hisSCDensity;
112  if (TString(inputHisto[0].data()).EndsWith(".root") && inputHisto[1].size() != 0) {
113  auto fileSCInput = std::unique_ptr<TFile>(TFile::Open(inputHisto[0].data()));
114  hisSCDensity.reset((TH3*)fileSCInput->Get(inputHisto[1].data()));
115  hisSCDensity->SetDirectory(nullptr);
116  }
117  if (distortionType == SpaceCharge::SCDistortionType::SCDistortionsConstant) {
118  LOG(INFO) << "TPC: Using constant space-charge distortions." << FairLogger::endl;
119  if (hisSCDensity == nullptr) {
120  LOG(FATAL) << "Constant space-charge distortions require an initial space-charge density histogram. Please provide the path to the root file (O2TPCSCDensityHisFilePath) and the histogram name (O2TPCSCDensityHisName) in your environment variables." << FairLogger::endl;
121  }
122  }
123  if (distortionType == SpaceCharge::SCDistortionType::SCDistortionsRealistic) {
124  LOG(INFO) << "TPC: Using realistic space-charge distortions." << FairLogger::endl;
125  }
126  if (hisSCDensity) {
127  LOG(INFO) << "TPC: Providing initial space-charge density histogram: " << hisSCDensity->GetName() << FairLogger::endl;
128  }
129 
130  mDigitizer.enableSCDistortions(distortionType, hisSCDensity.get(), gridSize[0], gridSize[1], gridSize[2]);
131  }
132  mDigitizer.setContinuousReadout(!triggeredMode);
133 
134  // setup the input chain for the hits
135  mSimChains.emplace_back(new TChain("o2sim"));
136 
137  // add the main (background) file
138  mSimChains.back()->AddFile(ic.options().get<std::string>("simFile").c_str());
139 
140  // maybe add a particular signal file
141  auto signalfilename = ic.options().get<std::string>("simFileS");
142  if (signalfilename.size() > 0) {
143  mSimChains.emplace_back(new TChain("o2sim"));
144  mSimChains.back()->AddFile(signalfilename.c_str());
145  }
146 
147  if (!gGeoManager) {
148  o2::Base::GeometryManager::loadGeometry();

152  void run(framework::ProcessingContext& pc)
153  {
154  static int callcounter = 0;
155  callcounter++;
156 
157  static bool finished = false;
158  if (finished) {
159  return;
160  }
161  LOG(INFO) << "Processing TPC digitization";
162 
163  // read collision context from input
164  auto context = pc.inputs().get<o2::steer::RunContext*>("collisioncontext");
165  auto& irecords = context->getEventRecords();
166  LOG(INFO) << "TPC: Processing " << irecords.size() << " collisions";
167  if (irecords.size() == 0) {
168  return;
169  }
170 
171  bool isContinuous = mDigitizer.isContinuousReadout();
172  if (mWriteGRP) {
173  auto roMode = isContinuous ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::PRESENT;
174  LOG(INFO) << "TPC: Sending ROMode= " << (mDigitizer.isContinuousReadout() ? "Continuous" : "Triggered")
175  << " to GRPUpdater from channel " << mChannel;
176  pc.outputs().snapshot(Output{ "TPC", "ROMode", 0, Lifetime::Timeframe }, roMode);
177  }
178 
179  // extract which sector to treat
180  auto const* sectorHeader = DataRefUtils::getHeader<TPCSectorHeader*>(pc.inputs().get("collisioncontext"));
181  if (sectorHeader == nullptr) {
182  LOG(ERROR) << "TPC: Sector header missing, skipping processing";
183  return;
184  }
185  auto sector = sectorHeader->sector;
186  LOG(INFO) << "TPC: Processing sector " << sector;
187  // the active sectors need to be propagated
188  uint64_t activeSectors = 0;
189  activeSectors = sectorHeader->activeSectors;
190 
191  // lambda that snapshots digits to be sent out; prepares and attaches header with sector information
192  auto snapshotDigits = [this, sector, &pc, activeSectors](std::vector<o2::TPC::Digit> const& digits) {
193  o2::TPC::TPCSectorHeader header{ sector };
194  header.activeSectors = activeSectors;
195  // note that snapshoting only works with non-const references (to be fixed?)
196  pc.outputs().snapshot(Output{ "TPC", "DIGITS", static_cast<SubSpecificationType>(mChannel), Lifetime::Timeframe,
197  header },
198  const_cast<std::vector<o2::TPC::Digit>&>(digits));
199  };
200  // lambda that snapshots labels to be sent out; prepares and attaches header with sector information
201  auto snapshotLabels = [this, &sector, &pc, activeSectors](o2::dataformats::MCTruthContainer<o2::MCCompLabel> const& labels) {
202  o2::TPC::TPCSectorHeader header{ sector };
203  header.activeSectors = activeSectors;
204  pc.outputs().snapshot(Output{ "TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(mChannel),
205  Lifetime::Timeframe, header },
206  const_cast<o2::dataformats::MCTruthContainer<o2::MCCompLabel>&>(labels));
207  };
208  // lambda that snapshots digits grouping (triggers) to be sent out; prepares and attaches header with sector information
209  auto snapshotEvents = [this, sector, &pc, activeSectors](const std::vector<DigiGroupRef>& events) {
210  o2::TPC::TPCSectorHeader header{ sector };
211  header.activeSectors = activeSectors;
212  LOG(INFO) << "TPC: Send TRIGGERS for sector " << sector << " channel " << mChannel << " | size " << events.size();
213  pc.outputs().snapshot(Output{ "TPC", "DIGTRIGGERS", static_cast<SubSpecificationType>(mChannel), Lifetime::Timeframe,
214  header },
215  const_cast<std::vector<DigiGroupRef>&>(events));
216  };
217 
218  std::vector<o2::TPC::Digit> digitsAccum; // accumulator for digits
219  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum; // timeframe accumulator for labels
220  std::vector<DigiGroupRef> eventAccum;
221 
222  // no more tasks can be marked with a negative sector
223  if (sector < 0) {
224  digitsAccum.clear();
225  labelAccum.clear();
226  std::vector<DigiGroupRef> evAccDummy;
227 
228  snapshotEvents(evAccDummy);
229  snapshotDigits(digitsAccum);
230  snapshotLabels(labelAccum);
231 
232  if (sector == -1) {
233  LOG(INFO) << "TPC: Processing done - exit through the gift shop...";
234  pc.services().get<ControlService>().readyToQuit(false);
235  finished = true;
236  }
237  return;
238  }
239 
240  mDigitizer.setSector(sector);
241  mDigitizer.init();
242 
243  auto& eventParts = context->getEventParts();
244 
245  auto flushDigitsAndLabels = [this, &digitsAccum, &labelAccum](bool finalFlush = false) {
246  // flush previous buffer
247  mDigits.clear();
248  mLabels.clear();
249  mDigitizer.flush(mDigits, mLabels, finalFlush);
250  LOG(INFO) << "TPC: Flushed " << mDigits.size() << " digits and " << mLabels.getNElements() << " labels";
251  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(digitsAccum));
252  labelAccum.mergeAtBack(mLabels);
253  };
254 
255  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
256  mDigitizer.setStartTime(sampaProcessing.getTimeBinFromTime(irecords[0].timeNS / 1000.f));
257 
258  TStopwatch timer;
259  timer.Start();
260 
261  // loop over all composite collisions given from context
262  // (aka loop over all the interaction records)
263  for (int collID = 0; collID < irecords.size(); ++collID) {
264  const float eventTime = irecords[collID].timeNS / 1000.f;
265  LOG(INFO) << "TPC: Event time " << eventTime << " us";
266  mDigitizer.setEventTime(eventTime);
267  if (!isContinuous) {
268  mDigitizer.setStartTime(sampaProcessing.getTimeBinFromTime(eventTime));
269  }
270  int startSize = digitsAccum.size();
271 
272  // for each collision, loop over the constituents event and source IDs
273  // (background signal merging is basically taking place here)
274  for (auto& part : eventParts[collID]) {
275  const int eventID = part.entryID;
276  const int sourceID = part.sourceID;
277 
278  // get the hits for this event and this source
279  std::vector<o2::TPC::HitGroup> hitsLeft;
280  std::vector<o2::TPC::HitGroup> hitsRight;
281  retrieveHits(mSimChains, getBranchNameLeft(sector).c_str(), part.sourceID, part.entryID, &hitsLeft);
282  retrieveHits(mSimChains, getBranchNameRight(sector).c_str(), part.sourceID, part.entryID, &hitsRight);
283  LOG(DEBUG) << "TPC: Found " << hitsLeft.size() << " hit groups left and " << hitsRight.size() << " hit groups right in collision " << collID << " eventID " << part.entryID;
284 
285  mDigitizer.process(hitsLeft, eventID, sourceID);
286  mDigitizer.process(hitsRight, eventID, sourceID);
287 
288  flushDigitsAndLabels();
289 
290  if (!isContinuous) {
291  eventAccum.emplace_back(startSize, digitsAccum.size() - startSize);
292  }
293  }
294  }
295 
296  // final flushing step; getting everything not yet written out
297  if (isContinuous) {
298  LOG(INFO) << "TPC: Final flush";
299  flushDigitsAndLabels(true);
300  eventAccum.emplace_back(0, digitsAccum.size()); // all digits are grouped to 1 super-event pseudo-triggered mode
301  }
302 
303  // send out to next stage
304  snapshotEvents(eventAccum);
305  snapshotDigits(digitsAccum);
306  snapshotLabels(labelAccum);
307 
308  timer.Stop();
309  LOG(INFO) << "TPC: Digitization took " << timer.CpuTime() << "s";
310  }

321 o2::framework::DataProcessorSpec getTPCDigitizerSpec(int channel, bool writeGRP)
322 {
323  // create the full data processor spec using
324  // a name identifier
325  // input description
326  // algorithmic description (here a lambda getting called once to setup the actual processing function)
327  // options that can be used for this processor (here: input file names where to take the hits)
328  std::stringstream id;
329  id << "TPCDigitizer" << channel;
330 
331  std::vector<OutputSpec> outputs; // define channel by triple of (origin, type id of data to be sent on this channel, subspecification)
332  outputs.emplace_back("TPC", "DIGITS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
333  outputs.emplace_back("TPC", "DIGTRIGGERS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
334  outputs.emplace_back("TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
335  if (writeGRP) {
336  outputs.emplace_back("TPC", "ROMode", 0, Lifetime::Timeframe);
337  LOG(INFO) << "TPC: Channel " << channel << " will supply ROMode";
338  }
339 
340  return DataProcessorSpec{
341  id.str().c_str(),
342  Inputs{ InputSpec{ "collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe } },
343  outputs,
344  AlgorithmSpec{ adaptFromTask<TPCDPLDigitizerTask>(channel, writeGRP) },
345  Options{ { "simFile", VariantType::String, "o2sim.root", { "Sim (background) input filename" } },
346  { "simFileS", VariantType::String, "", { "Sim (signal) input filename" } },
347  { "distortionType", VariantType::Int, 0, { "Distortion type to be used. 0 = no distortions (default), 1 = realistic distortions (not implemented yet), 2 = constant distortions" } },
348  { "gridSize", VariantType::String, "33,180,33", { "Comma separated list of number of bins in z, phi and r for distortion lookup tables (z and r can only be 2**N + 1, N=1,2,3,...)" } },
349  { "initialSpaceChargeDensity", VariantType::String, "", { "Path to root file containing TH3 with initial space-charge density and name of the TH3 (comma separated)" } },
350  { "TPCtriggered", VariantType::Bool, false, { "Impose triggered RO mode (default: continuous)" } } }
351  };
352 }
